01. SELECT E_ID,NAME,SALARY FROM EMPLOYEE GROUP BY SALARY ORDER BY SALARY DESC LIMIT 3;

02. SELECT E_ID,NAME,SALARY FROM EMPLOYEE WHERE SALARY IN (SELECT MIN(SALARY) FROM EMPLOYEE);

03. SELECT E.E_ID,E.NAME,D.DEPT_ID,D.DEPT_NAME FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.NAME = D.DEPT_MANAGER GROUP BY D.DEPT_ID ORDER BY D.DEPT_ID;

04. SELECT E.E_ID,E.NAME,D.DEPT_ID FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.NAME = D.DEPT_MANAGER GROUP BY D.DEPT_ID ORDER BY D.DEPT_ID;

05. SELECT E.E_ID,E.NAME,D.DEPT_ID,E.SALARY FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.NAME = D.DEPT_MANAGER HAVING MIN(E.SALARY);

06. SELECT E.E_ID,E.NAME,D.DEPT_ID,E.SALARY FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'COMMUNICATIONS';

07. SELECT E.E_ID,E.NAME,D.DEPT_ID,MAX(E.SALARY) FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'FINANCE';

08. SELECT E.E_ID,E.NAME,D.DEPT_ID,MIN(E.SALARY) FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'PRODUCT';

09. SELECT D.DEPT_NAME,COUNT(*) EMP_COUNT ,MAX(E.SALARY) FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'HEALTH';

10. SELECT E.E_ID,E.NAME FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_MANAGER = 'ROBERT SWIFT' AND E.NAME != D.DEPT_MANAGER;

11. SELECT COUNT(*) EMP_COUNT,D.DEPT_NAME,D.DEPT_MANAGER FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'HEALTH';

12. SELECT E.E_ID,D.DEPT_NAME,D.DEPT_ID,D.DEPT_MANAGER FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'COMMUNICATIONS';

13. SELECT D.DEPT_ID,D.DEPT_NAME,AVG(E.SALARY) AVG_EXPENSES FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID GROUP BY DEPT_ID;

14. SELECT D.DEPT_ID,D.DEPT_NAME,SUM(E.SALARY) TOTAL_EXPENSES FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_NAME = 'FINANCE';

15. SELECT D.DEPT_ID,D.DEPT_NAME,SUM(E.SALARY) TOTAL_EXPENSES FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID GROUP BY DEPT_ID ORDER BY TOTAL_EXPENSES LIMIT 1;

16. SELECT COUNT(*) EMP_COUNT, D.DEPT_NAME FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID GROUP BY DEPT_NAME;

17. SELECT COUNT(*) EMP_COUNT, D.DEPT_NAME FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE E.SALARY < 30000 GROUP BY DEPT_NAME ;

18. SELECT E.DEPT_ID,D.DEPT_NAME ,COUNT(*) EMP_COUNT FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_ID = 4;

19. SELECT COUNT(*) AS EMPLOYEE_COUNT, D.DEPT_NAME FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID GROUP BY D.DEPT_ID ORDER BY EMPLOYEE_COUNT DESC LIMIT 2;

20. SELECT E.E_ID,E.NAME FROM EMPLOYEE E INNER JOIN DEPARTMENT D ON E.DEPT_ID = D.DEPT_ID WHERE D.DEPT_MANAGER = 'TIM ARCHER';
